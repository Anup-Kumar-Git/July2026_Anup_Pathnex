   🔹Ansible - Install Apache and and start service
    -name: Install and Start Apache
     hosts: all
     become: yes
     tasks:
	-name: Install apache
	 yum:
	    name: httpd
	    state: present

	  
	-name: Start apache
	 service:
	     name: httpd
	     state: started
	     enabled: yes
