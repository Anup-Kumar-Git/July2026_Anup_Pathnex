🔹Ansible - Install & Enable Ngnix
 -name: Install and start Nginx on Pathnex
  hosts: all
  become: yes



  tasks:
     -name: Install ngnix
      yum:
	 name: ngnix
	 state: present


     -name: Enable nginx
      service:
	name: nginx
	state: started
	enabled: yes

           labels:
             app: pathnex-app
        spec:
           containers:
